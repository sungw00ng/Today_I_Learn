# 멀티프로세싱과 멀티스레딩
## 멀티프로세싱
- 문제가 생겨도 다른 프로세스 영향 X
- 격리성과 신뢰성 높음

## 멀티스레딩
- 가볍고 효율성이 높다.
- 문제가 생겨도 다른 프로세스 영향 O

## Java Code(2개의 외부 JVM)
```java
// File: SubProcess.java
import java.io.BufferedReader;
import java.io.InputStreamReader;
import java.net.HttpURLConnection;
import java.net.URL;

public class SubProcess {
    public static void main(String[] args) {
        try {
            String name = args.length > 0 ? args[0] : "이름 없음";

            URL url = new URL("https://w3schools.com/python/demopage.htm");
            HttpURLConnection conn = (HttpURLConnection) url.openConnection();
            conn.setRequestMethod("GET");

            BufferedReader in = new BufferedReader(
                    new InputStreamReader(conn.getInputStream()));

            StringBuilder sb = new StringBuilder();
            String inputLine;
            while ((inputLine = in.readLine()) != null) {
                sb.append(inputLine).append("\n");
            }

            in.close();
            conn.disconnect();

            long pid = ProcessHandle.current().pid();
            long parentPid = ProcessHandle.current().parent()
                    .map(ProcessHandle::pid).orElse(-1L);

            System.out.println("pid of parent: " + parentPid);
            System.out.printf("pid of %s: %d\n", name, pid);
            System.out.println(sb.toString());

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```

```java
// File: MainProcess.java
public class MainProcess {
    public static void main(String[] args) {
        System.out.println("pid of main: " + ProcessHandle.current().pid());

        try {
            ProcessBuilder pb1 = new ProcessBuilder("java", "SubProcess", "프로세스 1");
            ProcessBuilder pb2 = new ProcessBuilder("java", "SubProcess", "프로세스 2");

            pb1.inheritIO(); // 자식 프로세스의 출력을 현재 콘솔로
            pb2.inheritIO();

            Process p1 = pb1.start();
            p1.waitFor(); // 순차 실행. 병렬로 하고 싶으면 join 없이 start만

            Process p2 = pb2.start();
            p2.waitFor();

        } catch (Exception e) {
            e.printStackTrace();
        }
    }
}
```
